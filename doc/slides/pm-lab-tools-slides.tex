\subsection{Building}

\frame {

  \frametitle{Building}

  \texttt{cd pm-lab-tools}

  \texttt{git pull origin master \# Update from repository}

  \texttt{./build.sh client}

  \bigskip{}

  \texttt{build/pmlabclient i30pm1 12345 pmX \# Execute}

}
